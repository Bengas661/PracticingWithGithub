git branch branch_test


#Commits associated with the main branch

#Switch to the new branch
git checkout branch_test

# ** Creating and switching at the same time: **
  ** git checkout -b branch_test 


#Complete history (including main branch) is replicated when running:
git log --oneline

#  Update a file (i.e. notepad index.html -> git add. -> git commit -m "xxx xxx")
After the update:
git log --oneline  -> The new commit is not present in the master branch 

#Going back to main branch (after commiting)
git checkout main
